        //PROGRAM TO IMPLEMENTATION OF CIRCULAR QUEUE USING ARRAY

#include<stdio.h>
#include<conio.h>

int queue[10],frnt=0,rear=0,n=5, item;

int main()
{
    int choice;
    while(1)
    {
         printf("\nMenu:");
        printf("\n1.Add Element to Queue:");
        printf("\n2.Delete Element From Queue:");
        printf("\n3.Display:");
        printf("\nEnter Choice:");
        scanf("%d",&choice);
        switch(choice)
        {
            case 1: Add();
                    break;
            case 2: Delete1();
                    break;
            case 3: Display();
                    break;
            default:
                    printf("Wrong Input...PLZ TRY AGAIN:");
        }
    }
}
void Add()
{
    if((frnt==1&&rear==n)||(frnt==rear+1))
    {
        printf("\nQueue is Full--------OverFlow.\n");
        return;
    }

    if(frnt==0)
    {
        frnt=1;
        rear=1;
    }

    else if(rear==n)
    {
        rear=1;
    }
    else{
        rear=rear+1;
    }

int item;
printf("Enter Data:");
scanf("%d",&item);
queue[rear]=item;
    return;
}
void Delete1()
{
    if(frnt==0)
    {
        printf("\nQueue is Empty-- underFlow:\n");
        return;
    }
        item=queue[frnt];
        printf("\nElement Deleted:");

        if(frnt==rear)
        {
            frnt=0;
            rear=0;
        }

    else if(frnt==n)
    {
        frnt=1;
    }
    else
        {
            frnt=frnt+1;
        }
        return;
    }


void Display()
{
    int i;
    if(frnt==0)
    {
       printf("\n----Queue is Empty--Enter Element:\n");
        return;
    }
    if(frnt<=rear)
    {
    for(i=frnt;i<=rear;i++)
       {
           printf("\n%d",queue[i]);
       }
    }
    else
    {
        for(i=frnt;i<=n;i++)
        {
           printf("\n%d",queue[i]);
        }
        for(i=1;i<=rear;i++)
        {
           printf("\n%d",queue[i]);
        }

    }
}
